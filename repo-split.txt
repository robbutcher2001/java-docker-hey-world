## Migrate folder to new repo

git clone https://github.com/user/old-repo.git source-repo
cd source-repo
git remote rm origin
cat .git/config
git filter-branch --subdirectory-filter <directory> -- --all

# Only do if you need a subdirectory:
mkdir <directory 1>
mv * <directory 1>
git add .
git commit -m "Migrating to new repo with retained history"
# To here

git remote add origin https://github.com/user/target-repo.git
git pull origin master --allow-unrelated-histories
git push origin master


## Delete repo and reinitialise

rm -rf *
rm -rf .*
git init
# add a README.md or something
git add .
git commit -m 'Re-initialising repo'
git remote add origin <url>
git push --force --set-upstream origin master
